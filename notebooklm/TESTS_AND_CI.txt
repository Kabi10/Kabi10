name: ClickUp Integration

on:
  pull_request:
    types: [opened, synchronize, closed, reopened]
  pull_request_review:
    types: [submitted]

jobs:
  link-pr-to-clickup:
    runs-on: ubuntu-latest
    steps:
      - name: Extract ClickUp Task ID from PR
        id: extract_task_id
        run: |
          # Extract task ID from PR title (format: [CU-xxxxx] or similar)
          TITLE="${{ github.event.pull_request.title }}"
          BRANCH="${{ github.event.pull_request.head.ref }}"
          
          # Try to extract from title first
          TASK_ID=$(echo "$TITLE" | grep -oP '\[CU-[a-zA-Z0-9]+\]' | head -1 | tr -d '[]')
          
          # If not found in title, try branch name
          if [ -z "$TASK_ID" ]; then
            TASK_ID=$(echo "$BRANCH" | grep -oP 'CU-[a-zA-Z0-9]+' | head -1)
          fi
          
          echo "task_id=$TASK_ID" >> $GITHUB_OUTPUT
          echo "Found ClickUp Task ID: $TASK_ID"

      - name: Link PR to ClickUp Task
        if: steps.extract_task_id.outputs.task_id != ''
        uses: actions/github-script@v7
        with:
          script: |
            const taskId = '${{ steps.extract_task_id.outputs.task_id }}';
            const prUrl = context.payload.pull_request.html_url;
            const prNumber = context.payload.pull_request.number;
            const prTitle = context.payload.pull_request.title;
            
            console.log(`Linking PR #${prNumber} to ClickUp task ${taskId}`);
            console.log(`PR URL: ${prUrl}`);
            console.log(`PR Title: ${prTitle}`);
            
            // This logs the information that will be used by the webhook
            // The actual linking happens via ClickUp webhook (see setup guide)

      - name: Update PR Status Comment
        if: steps.extract_task_id.outputs.task_id != ''
        uses: actions/github-script@v7
        with:
          script: |
            const taskId = '${{ steps.extract_task_id.outputs.task_id }}';
            const prNumber = context.payload.pull_request.number;
            const action = context.payload.action;
            
            let statusMessage = '';
            switch(action) {
              case 'opened':
                statusMessage = `✅ PR linked to ClickUp task: **${taskId}**\n\nThis PR will automatically update the task status as it progresses.`;
                break;
              case 'closed':
                if (context.payload.pull_request.merged) {
                  statusMessage = `🎉 PR merged! Task **${taskId}** status should be updated to reflect completion.`;
                } else {
                  statusMessage = `❌ PR closed without merging. Task **${taskId}** status may need manual review.`;
                }
                break;
            }
            
            if (statusMessage) {
              github.rest.issues.createComment({
                issue_number: prNumber,
                owner: context.repo.owner,
                repo: context.repo.repo,
                body: statusMessage
              });
            }

  validate-pr-format:
    runs-on: ubuntu-latest
    steps:
      - name: Validate PR Title Format
        run: |
          TITLE="${{ github.event.pull_request.title }}"
          
          # Check if title contains ClickUp task ID
          if [[ ! "$TITLE" =~ \[CU-[a-zA-Z0-9]+\] ]]; then
            echo "⚠️  Warning: PR title should include ClickUp task ID in format [CU-xxxxx]"
            echo "Current title: $TITLE"
            echo "Recommended format: [CU-xxxxx] Brief description of changes"
          fi

      - name: Validate Branch Name Format
        run: |
          BRANCH="${{ github.event.pull_request.head.ref }}"
          
          # Check if branch contains ClickUp task ID
          if [[ ! "$BRANCH" =~ CU-[a-zA-Z0-9]+ ]]; then
            echo "⚠️  Warning: Branch name should include ClickUp task ID"
            echo "Current branch: $BRANCH"
            echo "Recommended format: feature/CU-xxxxx-brief-description"
          fi

  sync-pr-review-to-clickup:
    runs-on: ubuntu-latest
    if: github.event_name == 'pull_request_review'
    steps:
      - name: Extract Task ID and Sync Review
        id: sync_review
        run: |
          TITLE="${{ github.event.pull_request.title }}"
          BRANCH="${{ github.event.pull_request.head.ref }}"
          REVIEW_STATE="${{ github.event.review.state }}"
          REVIEWER="${{ github.event.review.user.login }}"
          
          # Extract task ID
          TASK_ID=$(echo "$TITLE" | grep -oP '\[CU-[a-zA-Z0-9]+\]' | head -1 | tr -d '[]')
          if [ -z "$TASK_ID" ]; then
            TASK_ID=$(echo "$BRANCH" | grep -oP 'CU-[a-zA-Z0-9]+' | head -1)
          fi
          
          echo "task_id=$TASK_ID" >> $GITHUB_OUTPUT
          echo "review_state=$REVIEW_STATE" >> $GITHUB_OUTPUT
          echo "reviewer=$REVIEWER" >> $GITHUB_OUTPUT
          
          echo "Review sync info:"
          echo "  Task ID: $TASK_ID"
          echo "  Review State: $REVIEW_STATE"
          echo "  Reviewer: $REVIEWER"

      - name: Log Review Sync
        if: steps.sync_review.outputs.task_id != ''
        run: |
          echo "Review from ${{ steps.sync_review.outputs.reviewer }} (${{ steps.sync_review.outputs.review_state }}) will be synced to task ${{ steps.sync_review.outputs.task_id }}"

name: PR Validation

on:
  pull_request:
    branches: [master, main]

jobs:
  validate-pr:
    runs-on: ubuntu-latest
    name: Validate Pull Request
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v4
        with:
          fetch-depth: 0
      
      - name: Check for deleted critical files
        run: |
          echo "🔍 Checking for deleted critical files..."
          
          # Get list of deleted files in this PR
          DELETED_FILES=$(git diff --name-status origin/${{ github.base_ref }}...HEAD | grep "^D" | awk '{print $2}')
          
          # Define critical files that should never be deleted
          CRITICAL_FILES=(
            "KNOWN_ISSUES.md"
            "QUICKSTART.md"
            "SETUP.md"
            "README.md"
            "ARCHITECTURE.md"
            "CONTRIBUTING.md"
            "ROADMAP.md"
            "app/google-services.json.template"
            "local.properties.template"
          )
          
          # Check if any critical files were deleted
          ERRORS=0
          for file in "${CRITICAL_FILES[@]}"; do
            if echo "$DELETED_FILES" | grep -q "^$file$"; then
              echo "❌ ERROR: Critical file '$file' was deleted!"
              ERRORS=$((ERRORS + 1))
            fi
          done
          
          # Check for deleted UI components
          if echo "$DELETED_FILES" | grep -q "app/src/main/java/com/senthapps/slagrimarket/ui/components/"; then
            echo "❌ ERROR: UI component files were deleted!"
            echo "$DELETED_FILES" | grep "app/src/main/java/com/senthapps/slagrimarket/ui/components/"
            ERRORS=$((ERRORS + 1))
          fi
          
          if [ $ERRORS -gt 0 ]; then
            echo ""
            echo "❌ PR validation failed: $ERRORS critical file(s) deleted"
            echo "Please restore these files or provide justification in the PR description."
            exit 1
          fi
          
          echo "✅ No critical files deleted"
      
      - name: Check PR size
        run: |
          echo "📊 Analyzing PR size..."
          
          # Get diff statistics
          STATS=$(git diff --shortstat origin/${{ github.base_ref }}...HEAD)
          echo "Changes: $STATS"
          
          # Extract numbers
          INSERTIONS=$(echo "$STATS" | grep -oP '\d+(?= insertion)' || echo "0")
          DELETIONS=$(echo "$STATS" | grep -oP '\d+(?= deletion)' || echo "0")
          
          echo "Insertions: $INSERTIONS"
          echo "Deletions: $DELETIONS"
          
          # Warn if PR is very large (>2000 lines changed)
          TOTAL=$((INSERTIONS + DELETIONS))
          if [ $TOTAL -gt 2000 ]; then
            echo "⚠️  WARNING: Large PR detected ($TOTAL lines changed)"
            echo "Consider breaking this into smaller PRs for easier review."
          fi
          
          # Error if massive deletions without corresponding additions
          if [ $DELETIONS -gt 1000 ] && [ $INSERTIONS -lt 100 ]; then
            echo "❌ ERROR: Suspicious PR - many deletions ($DELETIONS) with few additions ($INSERTIONS)"
            echo "This might indicate accidental file deletion. Please review carefully."
            exit 1
          fi
          
          echo "✅ PR size check passed"
      
      - name: Validate commit message
        run: |
          echo "📝 Validating commit messages..."
          
          # Get commit messages in this PR
          COMMITS=$(git log --pretty=format:"%s" origin/${{ github.base_ref }}...HEAD)
          
          # Check for empty commit messages
          if [ -z "$COMMITS" ]; then
            echo "❌ ERROR: No commits found in PR"
            exit 1
          fi
          
          echo "Commits in this PR:"
          echo "$COMMITS"
          echo ""
          echo "✅ Commit message validation passed"
      
      - name: Summary
        if: success()
        run: |
          echo "✅ All PR validation checks passed!"
          echo ""
          echo "This PR is ready for human review."

package com.senthapps.slagrimarket.data.repository

import com.senthapps.slagrimarket.data.api.ListingApiService

import kotlinx.coroutines.flow.first
import kotlinx.coroutines.flow.flowOf
import kotlinx.coroutines.flow.toList
import kotlinx.coroutines.test.runTest
import org.junit.Assert.*
import org.junit.Before
import org.junit.Test
import retrofit2.Response

@OptIn(ExperimentalCoroutinesApi::class)
class ListingRepositoryTest {

    private lateinit var listingApiService: ListingApiService
    private lateinit var listingDao: ListingDao
    private lateinit var localOpDao: LocalOpDao
    private lateinit var moshi: Moshi
    private lateinit var storageRepository: StorageRepository
    private lateinit var repository: ListingRepository

    private val mockListings = listOf(
        Listing(
            id = "1",
            farmerId = "farmer1",
            cropType = "Tomatoes",
            quantity = 100.0,
            unit = "kg",
            pricePerUnit = 50.0,
            quality = QualityGrade.A,
            harvestDate = "2025-11-20",
            location = "Jaffna",
            isActive = true,
            createdAt = "2025-11-20T10:00:00Z",
            updatedAt = "2025-11-20T10:00:00Z"
        ),
        Listing(
            id = "2",
            farmerId = "farmer2",
            cropType = "Rice",
            quantity = 500.0,
            unit = "kg",
            pricePerUnit = 80.0,
            quality = QualityGrade.B,
            harvestDate = "2025-11-19",
            location = "Colombo",
            isActive = true,
            createdAt = "2025-11-19T10:00:00Z",
            updatedAt = "2025-11-19T10:00:00Z"
        )
    )

    @Before
    fun setup() {
        listingApiService = mockk(relaxed = true)
        listingDao = mockk(relaxed = true)
        localOpDao = mockk(relaxed = true)
        moshi = Moshi.Builder().build()
        storageRepository = mockk(relaxed = true)

        repository = ListingRepository(
            listingApiService,
            listingDao,
            localOpDao,
            moshi,
            storageRepository
        )
    }

    @Test
    fun `getAllActiveListingsFlow should return listings from DAO`() = runTest {
        // Given: DAO returns listings
        coEvery { listingDao.getAllActiveListingsFlow() } returns flowOf(mockListings)

        // When: getAllActiveListingsFlow is called
        val result = repository.getAllActiveListingsFlow().first()

        // Then: Should return listings
        assertEquals(2, result.size)
        assertEquals("Tomatoes", result[0].cropType)
    }

    @Test
    fun `getAllActiveListings should emit cached data first`() = runTest {
        // Given: DAO has cached data
        val listingsResponse = ListingsResponse(
            listings = mockListings,
            totalCount = mockListings.size,
            page = 1,
            totalPages = 1,
            hasNext = false,
            hasPrevious = false,
            lastUpdated = "2025-11-20T10:00:00Z"
        )
        coEvery { listingDao.getAllActiveListings() } returns mockListings
        coEvery { listingApiService.getListings(any(), any()) } returns Response.success(listingsResponse)

        // When: getAllActiveListings is called
        val results = repository.getAllActiveListings().toList()

        // Then: Should emit loading, then cached data
        assertTrue(results.any { it is Resource.Loading })
        assertTrue(results.any { it is Resource.Success && it.data?.size == 2 })
    }

    @Test
    fun `createListing should save to DAO and create local op`() = runTest {
        // Given: DAO operations succeed
        coEvery { listingDao.insertListing(any()) } just Runs
        coEvery { localOpDao.insertOp(any()) } just Runs

        // When: createListing is called
        val result = repository.createListing(
            cropType = "Tomatoes",
            quantity = 100.0,
            unit = "kg",
            pricePerUnit = 50.0,
            quality = "A",
            harvestDate = "2025-11-20",
            location = "Jaffna",
            farmerId = "farmer1"
        )

        // Then: Should succeed and save to DAO
        assertTrue(result.isSuccess)
        coVerify { listingDao.insertListing(any()) }
        coVerify { localOpDao.insertOp(any()) }
    }

    @Test
    fun `updateListing should update DAO and create local op`() = runTest {
