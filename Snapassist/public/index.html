        el("btnList").onclick = async () => {
            const deviceId = el("deviceSelect").value;
            if (!deviceId) return log("Select a device first.", "err");
            
            try {
                await requestAppReport({ deviceId });
                log(`📋 LIST_APPS requested for ${deviceId}`, "ok");
                setTimeout(() => renderReport(deviceId), 1500);
            } catch (e) {
                log(`LIST_APPS failed: ${e.message}`, "err");
            }
        };

        async function renderReport(id) {
            try {
                const res = await getDoc(doc(db, `devices/${id}/reports/installed`));
                el("report").textContent = res.exists() ? 
                    JSON.stringify(res.data(), null, 2) : 
                    "(no report yet)";
            } catch (error) {
                log(`Error loading report: ${error.message}`, "err");
            }
        }
